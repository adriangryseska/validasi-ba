Template Validasi CS - Markaz Group Template Validasi Transaksi Tanggal Entri: Jenis Produk: Pulsa (REGULER) Data (INTERNET) Paket Telepon Nomor HP: Serial Number (SN): Jumlah (untuk Data): Durasi Hari (untuk Data): Buat Template Bersihkan Form Hasil Template: Salin Template function generateTemplate() { const tanggal = document.getElementById('tanggal').value || '[Tanggal N/A]'; const produk = document.getElementById('produk').value; const nomor = document.getElementById('nomor').value || '[Nomor N/A]'; const sn = document.getElementById('sn').value || '[SN N/A]'; const jumlah = document.getElementById('jumlah').value || '[Jumlah N/A]'; const durasi = document.getElementById('durasi').value || '[Durasi N/A]'; let template = ''; if (produk === 'REGULER') { template = `${tanggal}    TELKOMSEL REGULER ${nomor}    ${sn}  SUCCESS`; } else if (produk === 'DATA') { template = `${tanggal}    INTERNET DATA ${jumlah}GB ${durasi} HARI  ${nomor}    ${sn}  SUCCESS`; } else if (produk === 'TELEPON') { template = `Isi ulang paket Nelpon ${nomor} pd ${tanggal} berhasil Nomor seri ${sn}.`; } document.getElementById('output').textContent = template; } function clearInputs() { document.getElementById('tanggal').value = ''; document.getElementById('produk').selectedIndex = 0; document.getElementById('nomor').value = ''; document.getElementById('sn').value = ''; document.getElementById('jumlah').value = ''; document.getElementById('durasi').value = ''; document.getElementById('output').textContent = ''; } function copyOutput() { const output = document.getElementById('output').textContent; navigator.clipboard.writeText(output).then(() => { alert('Template berhasil disalin!'); }, () => { alert('Gagal menyalin template.'); }); }
